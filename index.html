Cole e execute o código no SQL Editor do painel Supabase para criar a tabela compatível: -- Opção A: Se você já tem a tabela chat_backups, execute este ALTER para adicionar a coluna user_id: ALTER TABLE public.chat_backups ADD COLUMN IF NOT EXISTS user_id uuid DEFAULT auth.uid(); -- Opção B: Criar a tabela do zero se ainda não existir: CREATE TABLE IF NOT EXISTS public.chat_backups ( id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY, created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL, user_id uuid DEFAULT auth.uid(), role text, content text, chat_html text, history_json json, settings_json json ); -- Ativa Segurança de Nível de Linha (RLS) se ainda não estiver ativa ALTER TABLE public.chat_backups ENABLE ROW LEVEL SECURITY; -- Garante política permissiva de acesso para garantir sincronização do usuário DROP POLICY IF EXISTS "Permitir acesso total para testes" ON public.chat_backups; CREATE POLICY "Permitir acesso total para testes" ON public.chat_backups FOR ALL TO anon, authenticated USING (true) WITH CHECK (true);
